### Line Highlighting

```js
１行
function HighlightText(highlight) {
  if (highlight) {
    // highlight-next-line
    return "This text is highlighted!";
  }

  return "Nothing highlighted";
}

複数行
// highlight-start
function HighlightText(highlight) {
  if (highlight) {
    return "This range is highlighted!";
  }

  return "Nothing highlighted";
}
// highlight-end
```
